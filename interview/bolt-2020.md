const isValidString = (str, count = 0) => {
  if (str.length === 0) return true;
  
  const map = {};
  const strArr = str.split('');
  
  for(let i = 0; i < strArr.length; i++) {
    if (str[i] in map) {
      map[str[i]] ++;
    } else {
      map[str[i]] = 1;
    }
  }
  
  let freq;
  let prev;
  
  for (let char in map) {
    if (!freq) {
      freq = map[char];
    } else {
      if (map[char] !== freq) {
